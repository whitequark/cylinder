Module Timestamp.Delta

module Delta: sig .. end
Timestamp deltas.
type t = private int64
Type of timestamp deltas. Timestamp deltas are represented by differences in Unix times in milliseconds.
val zero : t
zero is a delta corresponding to no difference in time.
val t_from_protobuf : Protobuf.Decoder.t -> t
t_from_protobuf d deserializes a timestamp delta from d.
val t_to_protobuf : t -> Protobuf.Encoder.t -> unit
t_to_protobuf t e serializes timestamp delta t into e.
val of_milliseconds : int64 -> t
of_milliseconds dut converts Unix time difference dut in milliseconds to a timestamp delta.
val to_milliseconds : t -> int64
to_milliseconds dt converts a timestamp delta dt to Unix time difference in milliseconds.
val to_string : t -> string
to_string dt converts a timestamp delta dt to a formatted string.
val add : t -> t -> t
add a b adds timestamp deltas a and b.
val sub : t -> t -> t
sub a b subtracts timestamp delta b from a.
val div : t -> int -> t
div dt n divides timestamp delta dt to n periods.
